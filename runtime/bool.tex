
\def\STR_bool{bool}%

\def\@pytexRuntime@bool#1{% -> bool object ptr
	\@pytexRuntime@allocType{bool}%
	\e\e\e\def\e\csname\@pytexRuntime@return @value\endcsname{#1}%
}

\def\@pytexRuntime@boolTrue{%
	\@pytexRuntime@bool{1}%
}

\def\@pytexRuntime@boolFalse{%
	\@pytexRuntime@bool{0}%
}

% #1 = bool object ptr
\def\@pytexRuntime@boolCopy#1{%
	\e\e\e\@pytexRuntime@bool\e\e\e{\csname#1@value\endcsname}%
}

\def\@pytexRuntime@boolIf#1{%
	\e\e\e\ifnum\csname#1@value\endcsname=1 %
}
