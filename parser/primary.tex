
% await_primary:
%     | 'await' primary 
%     | primary
\def\@pytexParser@tryAwaitPrimary{%
    \@pytexParser@choice%
    {%
        \@pytexParser@join%
            {\@pytexParser@tryToken{\tokentypeAWAIT}}%
            {\@pytexParser@tryPrimary}%
            {\e\def\e\@pytexTMP@parserReturnValue\e{\e\@pytexOperator@await\e{\second}}}%
    }%
    {\@pytexParser@tryPrimary}%
}
